import collections
from collections import Counter
import operator
damagePerSpecies = dict(Counter(dataset['Species Name']))
#x = sorted(damagePerSpecies.items(), key=operator.itemgetter(0))
damagePerSpecies = sorted(damagePerSpecies.items(), key=lambda x: x[1], reverse=True)[0:10]
species = []
damageVal = []
for i in damagePerSpecies:
    species.append(i[0])
    damageVal.append(i[1])
    
import seaborn as sns
%matplotlib inline
fig=plt.figure(figsize=(20,5))
ax1=fig.add_subplot(111)
plt.xticks(rotation=90)
sns.barplot(x=species, y=damageVal,ax=ax1)
